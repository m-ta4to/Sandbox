function caseInSwitch(val) {
  let answer = "";
  // Змініть код лише під цим рядком
  switch(val) {
    case 1:
    answer = "alpha";
    break;
    case 2:
    answer="beta";
    break;
    case 3:
    answer = "gamma";
    break;
    case 4:
    answer = "delta";
    break;

  }



  // Змініть код лише над цим рядком
  return answer;
}

caseInSwitch(1);
